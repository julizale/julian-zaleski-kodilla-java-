
SELECT I.ID AS ISSUE_ID, I.SUMMARY, I.DESCRIPTION, IL.NAME AS LIST_NAME
FROM ISSUES I JOIN ISSUESLISTS IL ON I.ISSUESLIST_ID = IL.ID;

SELECT I.ID AS ISSUE_ID, I.SUMMARY, I.DESCRIPTION,
       U.FIRSTNAME AS USER_FIRSTNAME, U.LASTNAME AS USER_LASTNAME
FROM ISSUES I JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
ORDER BY I.ID;

SELECT U.LASTNAME, U.FIRSTNAME, COUNT(*) AS ISSUES_ASSIGNED
FROM USERS U JOIN ISSUES I ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY I.USER_ID_ASSIGNEDTO
ORDER BY U.LASTNAME, U.FIRSTNAME;
